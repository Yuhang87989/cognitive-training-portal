// ====== AI智能回复（支持DeepSeek API） ======
// 优先使用真实API，失败则fallback到本地智能回复

async function callDeepSeek(prompt) {
    // 获取API Key（优先使用用户保存的，否则使用默认Key）
    const savedApiKey = localStorage.getItem('deepseek_api_key');
    const apiKey = savedApiKey || 'sk-8413f72a3f084fb08c84389555a76d37';
    
    try {
        // 尝试调用真实DeepSeek API
        const response = await fetch('https://api.deepseek.com/v1/chat/completions', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${apiKey}`
            },
            body: JSON.stringify({
                model: 'deepseek-chat',
                messages: [
                    {
                        role: 'system',
                        content: '你是一位专业的教育助手，擅长为中学生解答数学、语文、英语、物理、化学等学科问题。请用简洁清晰的语言回答，并给出具体的解题思路和学习建议。'
                    },
                    {
                        role: 'user',
                        content: prompt
                    }
                ],
                temperature: 0.7,
                max_tokens: 1000
            })
        });
        
        if (response.ok) {
            const data = await response.json();
            if (data.choices && data.choices[0] && data.choices[0].message) {
                return data.choices[0].message.content;
            }
        }
        
        // API调用失败，记录错误并fallback到本地
        console.log('DeepSeek API调用失败，使用本地智能回复');
    } catch (error) {
        console.log('DeepSeek API调用异常:', error.message);
    }
    
    // ====== 本地智能回复fallback ======
    await new Promise(r => setTimeout(r, 800));
    
    const q = prompt.toLowerCase();
    
    // 数学相关
    if (q.includes('数学') || q.includes('计算') || q.includes('方程') || q.includes('分数') || q.includes('几何')) {
        return '📐 数学问题分析：\n\n' + getMathResponse(prompt);
    }
    // 语文相关
    if (q.includes('语文') || q.includes('阅读') || q.includes('作文') || q.includes('古诗词') || q.includes('文言文')) {
        return '📝 语文问题分析：\n\n' + getChineseResponse(prompt);
    }
    // 英语相关
    if (q.includes('英语') || q.includes('english') || q.includes('语法') || q.includes('单词')) {
        return '📖 英语问题分析：\n\n' + getEnglishResponse(prompt);
    }
    // 物理相关
    if (q.includes('物理') || q.includes('力学') || q.includes('电学') || q.includes('光学')) {
        return '⚡ 物理问题分析：\n\n' + getPhysicsResponse(prompt);
    }
    // 化学相关
    if (q.includes('化学') || q.includes('元素') || q.includes('方程式')) {
        return '🧪 化学问题分析：\n\n' + getChemistryResponse(prompt);
    }
    // 批改相关
    if (q.includes('批改') || q.includes('学生答案')) {
        return '✏️ 批改结果：\n\n' + getCorrectResponse(prompt);
    }
    // 解说相关
    if (q.includes('解说') || q.includes('讲解') || q.includes('分析')) {
        return '💡 题目解说：\n\n' + getExplainResponse(prompt);
    }
    
    // 默认回复
    return '好的，我理解你的问题。让我为你分析：\n\n这是一个很好的问题，建议你：\n1. 仔细审题，找出关键信息\n2. 回顾相关知识点\n3. 尝试从不同角度思考\n4. 如果还有疑问，可以继续问我！';
}
